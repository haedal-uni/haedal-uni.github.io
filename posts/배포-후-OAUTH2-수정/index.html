function oauth(){
    let userInfoUrl
    let kakakokEY
    $.ajax({
        type: "GET",
        url: host + `/kakao/login`,
        contentType: false,
        success: function (response) {
            userInfoUrl = response[0];
            kakakokEY = response[1];

            Kakao.init(kakakokEY);
            Kakao.Auth.authorize({
                redirectUri: userInfoUrl,
            });
        }
    })
}
